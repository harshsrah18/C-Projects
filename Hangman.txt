// Function to choose a random word from the list
string getRandomWord() {
    vector<string> words = {"programming", "hangman", "computer", "development", "software", "game", "puzzle"};
    srand(time(0)); // Seed for random number generation
    int index = rand() % words.size();
    return words[index];
}

// Function to display the word with guessed letters
void displayWord(const string& word, const string& guessedLetters) {
    for (char c : word) {
        if (guessedLetters.find(c) != string::npos) {
            cout << c << " ";
        } else {
            cout << "_ ";
        }
    }
    cout << endl;
}

// Function to check if the user has guessed all letters
bool isWordGuessed(const string& word, const string& guessedLetters) {
    for (char c : word) {
        if (guessedLetters.find(c) == string::npos) {
            return false; // A letter is missing
        }
    }
    return true; // All letters are guessed
}
